COURSES :::

    2. Templates 
    3. Databases and Admin Panel

# TO START NEW PROJECT 
pip install pipenv
django-admin startproject myproject
    # -> settings.py -> Add and remove configuration
    # -> urls.py -> Add urls to run a webpage

# TO RUN THE PROJECT
python manage.py migrate
py manage.py runserver

# TO START NEW APP
py startapp myapp
    # -> admin.py -> To change data to admin users
    # -> models.py -> To create database models 
    # -> views.py -> To view the rendering app 
    # -> tests.py -> To run testcases

# TO CREATE AND INTERACT WITH DATABASE
py manage.py createsuperuser
# Give username,email,password and go to 8000/admin/auth
